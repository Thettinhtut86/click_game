# 1. Check Erlang
erl -eval 'io:format("Erlang version: ~s~n", [erlang:system_info(otp_release)]), halt().' -noshell
# Should output: Erlang version: 25

# 2. Check RabbitMQ
rabbitmqctl status | head -5

# 3. Check RabbitMQ is listening
ss -tlnp | grep 5672
# Should show: LISTEN *:5672

# 4. Enable management UI
rabbitmq-plugins enable rabbitmq_management
systemctl restart rabbitmq-server

# 5. Check management UI is listening
ss -tlnp | grep 15672
# Should show: LISTEN *:15672
